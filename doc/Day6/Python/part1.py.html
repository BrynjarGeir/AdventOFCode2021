import numpy as np

txt = 'input'
with open(txt) as f:
    line = f.readline()
    line = line.split(',')
    for idx, item in enumerate(line):
        line[idx] = int(item)

fishes = np.array(line)

def oneDay(fishes):
    tmp = np.array([], dtype=int)
    for idx, val in enumerate(fishes):
        if val == 0:
            fishes[idx] = 6
            tmp = np.append(tmp, 8)
        else:
            fishes[idx] -= 1
    fishes = np.append(fishes,tmp)
    return fishes
